  def session_token
    url =  URI::HTTPS.build({:host => GOOGLE_HOST_URL,
        :path => GOOGLE_AUTHSUB_SESSION_TOKEN_PATH})   
    begin
     @token = get(url).body.match(/^Token=(.*)$/)[1]
   rescue
     raise "ERROR: Invalid session token response."
   end
  end
