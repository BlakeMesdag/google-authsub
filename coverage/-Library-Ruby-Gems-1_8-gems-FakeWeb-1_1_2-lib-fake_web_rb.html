181     def baked_response
182       resp = case options[:response]
183       when Net::HTTPResponse: options[:response]
184       when String
185         socket = Net::BufferedIO.new(options[:response])
186         r = Net::HTTPResponse.read_new(socket)
187         r.instance_eval { @header['transfer-encoding'] = nil }
